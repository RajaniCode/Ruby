Answer:

class ListingZombiesTest < ActionDispatch::IntegrationTest
  setup { @user = User.create! }

  test 'valid token lists zombies' do
    get '/zombies', {}, {'Authorization'  => "Token token=#{@user.auth_token}"}
  end
end
