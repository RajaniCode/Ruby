Refactor this code to use send in place of the direct calls to list, emulate, and find. Hint: For the emulate and find methods that take parameters, you can pass those after the method's name when using send.

library.rb
class Library
  attr_accessor :games
 
  def initialize(games)
    self.games = games
  end
 
  def list
    puts games.join("\n")
  end
 
  def emulate(name)
    game = find(name)
    puts "Starting emulator for #{game}..."
  end
 
  def find(name)
    games.detect { |game| game.name == name }
  end
end


main.rb
library = Library.new(GAMES)
library.list
library.emulate("Contra")
game = library.find("Contra")
